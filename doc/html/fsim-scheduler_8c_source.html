82 u8 xe_scheduler_take_most_fsim_faithful (ocrScheduler_t *base, struct _ocrCost_t *cost, u32 *count,
83  ocrGuid_t *edts, struct _ocrPolicyCtx_t *context) {
84  ocrSchedulerHc_t* hcDerived = (ocrSchedulerHc_t*) base;
85  ocrWorker_t* w = NULL;
86  ocrGuid_t wid = context->sourceObj;
87  deguidify(getCurrentPD(), wid, (u64*)&w, NULL);
88 
89  ocrGuid_t popped = NULL_GUID;
90  ocrWorkpile_t * wp_to_pop = NULL;
91 
92  u64 id = get_worker_id(w);
93  if (id >= hcDerived->worker_id_begin && id <= hcDerived->worker_id_end) {
94  // XE worker trying to extract 'executable work' from CE assigned workpile
95  wp_to_pop = xe_scheduler_pop_mapping_to_assigned_work(base, w);
96  // TODO sagnak this is true because XE and CE does not touch the workpile simultaneously
97  // XE is woken up after the CE is done pushing the work here
98  // TODO sagnak, just to get it to compile, I am trickling down the 'cost' though it most probably is not the same
99  popped = wp_to_pop->fctPtrs->pop(wp_to_pop, cost);
100  } else {
101  // CE worker trying to extract buffered 'executable work' that XE encountered
102  // TODO sagnak NOT IDEAL, and the XE may be simultaneously pushing, therefore we 'steal' for synchronization
103  wp_to_pop = xe_scheduler_pop_mapping_to_work_shipping(base, w);
104  // TODO sagnak, just to get it to compile, I am trickling down the 'cost' though it most probably is not the same
105  popped = wp_to_pop->fctPtrs->steal(wp_to_pop, cost);
106  }
107 
108  *count = 1;
109  edts[0] = popped;
110  return 0;
111 }

113 // XE scheduler give can be called from three sites:
114 // first being the ocrEdtSchedule, pushing user created work when executing user code
115 // those tasks should end up in the workpile designated for buffering work for the CE to take
116 // the second being, XE pushing a 'message task' which should be handed out to the CE policy domain
117 // third being the CE worker pushing work onto 'executable task' workpile
118 // the differentiation between {1,2} and {3} is through the worker id
119 // the differentiation between {1} and {2} is through a runtime check of the underlying task type
120 u8 xe_scheduler_give_fsim_faithful (ocrScheduler_t* base, u32 count, ocrGuid_t* edts, struct _ocrPolicyCtx_t *context ) {
121  ocrSchedulerHc_t* hcDerived = (ocrSchedulerHc_t*) base;
122  ocrWorker_t* w = NULL;
123  ocrGuid_t wid = context->sourceObj;
124  deguidify(getCurrentPD(), wid, (u64*)&w, NULL);
125 
126  ocrTaskFsimBase_t* task = NULL;
127  u8 i = 0;
128  for ( ; i < count; ++i ) {
129  ocrGuid_t tid = edts[i];
130  deguidify(getCurrentPD(), tid, (u64*)&task, NULL);
131 
132  fsim_message_interface_t* taskAsMessage = &(task->message_interface);
133 
134  u64 id = get_worker_id(w);
135  if (id >= hcDerived->worker_id_begin && id <= hcDerived->worker_id_end) {
136  // if the pusher is an XE worker
137  if ( !taskAsMessage->is_message(taskAsMessage) ) {
138  // TODO sagnak Multiple XE push mappings, do not use the class push_mapping indirection
139  ocrWorkpile_t * wp_to_push = xe_scheduler_push_mapping_to_work_shipping(base, w);
140  wp_to_push->fctPtrs->push(wp_to_push,tid);
141 
142  // TODO sagnak, this assumes (*A LOT*) the structure below, is this fair?
143  // no assigned work found, now we have to create a 'message task'
144  // by using our policy domain's message task factory
145  ocrPolicyDomain_t* policy_domain = base->domain;
146  ocrTaskFactory_t* message_task_factory = policy_domain->taskFactories[1];
147 
148  // the message to the CE says 'give me work' and notes who is asking for it
149  ocrTaskFsimMessage_t* derivedMessage = (ocrTaskFsimMessage_t*)
150  message_task_factory->instantiate(message_task_factory, NULL, NULL, NULL, 0, NULL);
151  guidify(getCurrentPD(), (u64)task, &(task->guid), OCR_GUID_EDT);
152  ocrGuid_t messageTaskGuid = task->guid;
153  derivedMessage -> type = PICK_MY_WORK_UP;
154  derivedMessage -> from_worker_guid = wid;
155 
156  // give the work to the XE scheduler, which in turn should give it to the CE
157  // through policy domain hand out and the scheduler differentiates tasks by type (RTTI) like
158  // used to be 'base->give(base, wid, messageTaskGuid);'
159  // TODO sagnak, taking the address of a stack pointer, may not be the best idea
160  // TODO sagnak, fix the context value avoidance and set the source to 'wid'
161  xe_scheduler_give_fsim_faithful(base, 1, &messageTaskGuid, NULL);
162  } else {
163  ocrPolicyDomain_t* xePolicyDomain = base->domain;
164  xePolicyDomain->handOut(xePolicyDomain, wid, tid);
165  }
166  } else {
167  // if the pusher is a CE worker
168  ocrWorkpile_t * wp_to_push = xe_scheduler_push_mapping_to_assigned_work(base, w);
169  wp_to_push->fctPtrs->push(wp_to_push,tid);
170  }
171  }
172  return 0;
173 }

273 // this can be called from three different sites:
274 // one being the initial(from master) work that CE should dissipate
275 // other being the XEs giving a 'message task' for the CE to be notified
276 // last being the CEs giving a 'message task' to itself if it can not serve the message
277 u8 ce_scheduler_give (ocrScheduler_t* base, u32 count, ocrGuid_t* edts, struct _ocrPolicyCtx_t *context ) {
278  ocrWorker_t* w = NULL;
279  ocrGuid_t wid = context->sourceObj;
280  deguidify(getCurrentPD(), wid, (u64*)&w, NULL);
281 
282  u32 i = 0;
283  for ( ; i < count; ++i ) {
284  ocrGuid_t taskGuid = edts[i];
285  ocrTaskFsimBase_t* task = NULL;
286  deguidify(getCurrentPD(), taskGuid, (u64*)&task, NULL);
287 
288  fsim_message_interface_t* taskAsMessage = &(task->message_interface);
289 
290  ocrWorkpile_t * workpileToPush = NULL;
291 
292  if ( !taskAsMessage->is_message(taskAsMessage) ) {
293  workpileToPush = ce_scheduler_push_mapping_to_work(base, w);
294  } else {
295  // TODO sagnak Multiple XE push mappings, do not use the class push_mapping indirection
296  // TODO this is some foreign XE worker, what is the point in trickling down the worker id?
297  // TODO sagnak this should be a LOCKED data structure
298  // TODO there is no way to pick which 'message task pool' to go to for now :(
299  workpileToPush = ce_scheduler_push_mapping_to_messages(base);
300  }
301 
302  workpileToPush->fctPtrs->push(workpileToPush,taskGuid);
303  }
304  return 0;
305 }
